

>>>> CURRENCY_PAIR_PER_TIMEFRAME Data

time		: datetime index for the corresponding timeframe

ask_open	: open price for buying 

bid_open	: open price for selling

ask_close	: close price for selling

bid_close	: close price for buying

ask_high	: highest ask price among all ticks in one timeframe unit

bid_high	: highest bid price among all ticks in one timeframe unit

ask_low		: lowest ask price among all ticks in one timeframe unit

bid_low		: lowest bid price among all ticks in one timeframe unit

spread_open	: spread at open

spread_close	: spread at close

ask_volume	: buying volume in units

bid_volume	: selling volume in units

tick_volume	: number of ticks in one timeframe unit

correct_trades		: 'buy' if (df['bid'].idxmax()-df['bid'].idxmin()).days>=0 and (bid_high-ask_low) > 0 
			  'sell' if (df['bid'].idxmin()-df['bid'].idxmax()).days>=0 and (bid_high-ask_low) > 0
			  else: 'hold'

max_rets		: [(bid_high-ask_low) / ask_low] if the trade is buy and [(bid_high-ask_low) / bid_high] if the trade is sell

min_rets		: [-(bid_high-ask_low) / bid_high] if the trade is buy and [-(bid_high-ask_low) / ask_low] if the trade is sell 

naieve_buy_rets		: (bid_close-ask_open) / ask_open

naieve_sell_rets	: (bid_open-ask_close) / bid_open

boarder_correct_trades	: 'buy' if (df['bid'].idxmax()-df['bid'].idxmin()).days>=0 and (bid_close-ask_open) > 0 
			 'sell' if (df['bid'].idxmin()-df['bid'].idxmax()).days>=0 and (bid_open-ask_close) > 0
			 else: 'NA'

boarder_max_rets	: [(bid_close-ask_open) / ask_open] if the trade is buy and [(bid_open-ask_close) / bid_open] if the trade is sell

boarder_min_rets	: [-(bid_open-ask_close) / bid_open] if the trade is buy and [-(bid_close-ask_open) / ask_open] if the trade is sell 


>>>> STATS_PER_TIMEFRAME Data

comp_min_rets		: compounded returns of min_rets series

comp_naieve_buy_rets	: compounded returns of naieve_buy_rets series

comp_naieve_sell_rets	: compounded returns of naieve_sell_rets series

avg_max_rets		: average returns of max_rets series

avg_min_rets		: average returns of max_rets series

avg_naieve_buy_rets	: average returns of naieve_buy_rets series

avg_naieve_sell_rets	: average returns of naieve_sell_rets series

max_vol			: volatility of max_rets series

min_vol			: volatility of min_rets series

naieve_buy_vol		: volatility of naieve_buy_rets series

naieve_sell_vol		: volatility of naieve_sell_rets series

comp_max_vol		: compounded volatility of max_rets series

comp_min_vol		: compounded volatility of min_rets series

comp_naieve_buy_vol	: compounded volatility of naieve_buy_rets series

comp_naieve_sell_vol	: compounded volatility of naieve_sell_rets series

max_ror			: return on risk of max_rets series

min_ror			: return on risk of min_rets series

naieve_buy_ror		: return on risk of naieve_buy_rets series

naieve_sell_ror		: return on risk of naieve_sell_rets series

max_sharp_ror		: adjusted return on risk (sharp satio) of max_rets series

min_sharp_ror		: adjusted return on risk (sharp satio) of min_rets series

naieve_buy_sharp_ror	: adjusted return on risk (sharp satio) of naieve_buy_rets series

naieve_sell_sharp_ror	: adjusted return on risk (sharp satio) of naieve_sell_rets series

max_JB			: Jarque Bera's normality test score (p_value) of max_rets series, the closer to 0 the more normal the distribution  

min_JB			: Jarque Bera's normality test score (p_value) of min_rets series, the closer to 0 the more normal the distribution  

naieve_buy_JB		: Jarque Bera's normality test score (p_value) of naieve_buy_rets series, the closer to 0 the more normal the distribution  

naieve_sell_JB		: Jarque Bera's normality test score (p_value) of naieve_sell_rets series, the closer to 0 the more normal the distribution  

max_var			: value at risk of max_rets series

min_var			: value at risk of min_rets series

naieve_buy_var		: value at risk of naieve_buy_rets series

naieve_sell_var		: value at risk of naieve_sell_rets series
